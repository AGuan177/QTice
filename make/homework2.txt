#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int a[100000][10];

int main() {
int i,n,j=1;
double k1,k2,m1,m2,t[10000],mint=10000000,left=-10000,right=10000;
scanf("%d",&n);
for(i=1;i<=n;i++){
  scanf("%d %d",&a[i][1],&a[i][2]);
}
while(right-left>=0.000000001){
	m1=0;
	m2=0;
	for(i=1;i<=n;i++){
		k1=sqrt(pow(abs(a[i][1]-left),2)+pow(abs(a[i][2]),2));
		k2=sqrt(pow(abs(a[i][1]-right),2)+pow(abs(a[i][2]),2));
		if(m1<k1) m1=k1;
		if(m2<k2) m2=k2;
	}
	if(m1<m2) {right=(10*right+left)/11;t[j]=m1;}
	else {left=(10*left+right)/11;t[j]=m2;}
	j++;
}
for(i=1;i<j;i++){
	if(mint>t[i]) mint=t[i];
}
printf("%.4lf\n",mint);
	return 0;
}
